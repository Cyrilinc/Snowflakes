 224:     public static function resizeImage($image, $width, $height, $scale)
 225:     {
 226:         list($imagewidth, $imageheight, $imageType) = getimagesize($image);
 227:         $imageType = image_type_to_mime_type($imageType);
 228:         $newImageWidth = ceil($width * $scale);
 229:         $newImageHeight = ceil($height * $scale);
 230: 
 231:         switch ($imageType)
 232:         {
 233:             case "image/gif":
 234:                 $source = imagecreatefromgif($image);
 235:                 break;
 236:             case "image/pjpeg":
 237:             case "image/jpeg":
 238:             case "image/jpg":
 239:                 $source = imagecreatefromjpeg($image);
 240:                 break;
 241:             case "image/png":
 242:             case "image/x-png":
 243:                 $source = imagecreatefrompng($image);
 244:                 break;
 245:         }
 246:         $newImage = imagecreatetruecolor($newImageWidth, $newImageHeight);
 247: 
 248: 
 249:         if (($imageType == "image/gif") || ($imageType == "image/png") || ($imageType == "image/x-png"))
 250:         {
 251: 
 252:             $colourTotal = imagecolorstotal($source);
 253:             imagetruecolortopalette($newImage, true, $colourTotal <= 0 ? 1 : $colourTotal);
 254:             $currentTransparent = imagecolortransparent($source);
 255: 
 256:             // If we have a specific transparent color
 257:             if ($currentTransparent >= 0)
 258:             {
 259: 
 260:                 // Get the original image's transparent color's RGB values
 261:                 $transparentColor = imagecolorsforindex($source, $currentTransparent);
 262: 
 263:                 // Allocate the same color in the new image resource
 264:                 $currentTransparent = imagecolorallocate($newImage, $transparentColor['red'], $transparentColor['green'], $transparentColor['blue']);
 265: 
 266:                 // Completely fill the background of the new image with allocated color.
 267:                 imagefill($newImage, 0, 0, $currentTransparent);
 268: 
 269:                 // Set the background color for new image to transparent
 270:                 imagecolortransparent($newImage, $currentTransparent);
 271:             }
 272:             // Always make a transparent background color for PNGs that don't have one allocated already
 273:             elseif (($imageType == "image/png") || ($imageType == "image/x-png"))
 274:             {
 275: 
 276:                 // Turn off transparency blending (temporarily)
 277:                 imagealphablending($newImage, false);
 278: 
 279:                 // Create a new transparent color for image
 280:                 $color = imagecolorallocatealpha($newImage, 0, 0, 0, 127);
 281: 
 282:                 // Completely fill the background of the new image with allocated color.
 283:                 imagefill($newImage, 0, 0, $color);
 284: 
 285:                 // Restore transparency blending
 286:                 imagesavealpha($newImage, true);
 287:             }
 288:         }
 289: 
 290:         imagecopyresampled($newImage, $source, 0, 0, 0, 0, $newImageWidth, $newImageHeight, $width, $height);
 291: 
 292:         switch ($imageType)
 293:         {
 294:             case "image/gif":
 295:                 imagegif($newImage, $image);
 296:                 break;
 297:             case "image/pjpeg":
 298:             case "image/jpeg":
